267 template <typename T = Model, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
268 T & register_model(const std::string & name, bool nonlinear = false, bool merge_input = true)
269 {
270 if (name == this->name())
271 throw SetupException("Model named '" + this->name() +
272 "' is trying to register itself as a sub-model. This is not allowed.");
273
274 OptionSet extra_opts;
275 extra_opts.set<NEML2Object *>("_host") = host();
276 extra_opts.set<bool>("_nonlinear_system") = nonlinear;
277
278 auto model = Factory::get_object_ptr<T>("Models", name, extra_opts);
279 if (std::find(_registered_models.begin(), _registered_models.end(), model.get()) !=
280 _registered_models.end())
281 throw SetupException("Model named '" + name + "' has already been registered.");
282
283 if (merge_input)
284 for (auto && [name, var] : model->input_variables())
285 clone_input_variable(*var);
286
287 _registered_models.push_back(model.get());
288 return *model;
289 }
