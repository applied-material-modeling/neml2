242template <bool scaled>
243NonlinearSystem::Res<scaled>
244NonlinearSystem::residual()
245{
246 Res<false> r;
247 assemble(&r, nullptr);
248 if constexpr (scaled)
249 return scale(r);
250 else
251 return r;
252}

254template <bool scaled>
255NonlinearSystem::Res<scaled>
256NonlinearSystem::residual(const NonlinearSystem::Sol<scaled> & x)
257{
258 set_guess(x);

260}

262template <bool scaled>
263NonlinearSystem::Jac<scaled>
264NonlinearSystem::Jacobian()
265{
266 Jac<false> J;
267 assemble(nullptr, &J);
268 if constexpr (scaled)
269 return scale(J);

282template <bool scaled>
283std::tuple<NonlinearSystem::Res<scaled>, NonlinearSystem::Jac<scaled>>
284NonlinearSystem::residual_and_Jacobian()
285{
286 Res<false> r;
287 Jac<false> J;
288 assemble(&r, &J);
289 if constexpr (scaled)
290 return {scale(r), scale(J)};
291 else
292 return {r, J};
293}
