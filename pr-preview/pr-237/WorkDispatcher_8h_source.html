245template <typename I, typename O, typename Of, typename Ip, typename Op>
246void
247WorkDispatcher<I, O, Of, Ip, Op>::init_thread_pool()
248{
249 if (!_async)
250 return;
251
252 // Setup the task queue
253 for (const auto & device : _devices)
254 _tasks[device] = std::queue<std::function<void()>>();
255
256 auto nthread = _devices.size();
257 _thread_pool.reserve(nthread);
258 for (std::size_t i = 0; i < nthread; ++i)
259 {
260 // This is necessary to initialize the torch linear algebra library prior to threaded calls
261 // See: https://github.com/pytorch/pytorch/issues/90613
262 auto res = R2::identity().to(_devices[i]).inverse();
263 _thread_pool.emplace_back([this, i] { thread_pool_main(_devices[i]); });
264 }
265
266 // Initialize the thread
267 if (_thread_init)
268 {
269 for (std::size_t i = 0; i < nthread; ++i)
270 {
271 auto device = _devices[i];
272 auto task = [this, device = device]() mutable
273 {
274 _thread_init(device);
275 _scheduler.completed_work(device, 1);
276 };
277 _scheduler.dispatched_work(device, 1);
278 {
279 std::lock_guard<std::mutex> lock(_qmutex);
280 _tasks.at(device).push(task);
281 }
282 _thread_condition.notify_all();
283 }
284 _scheduler.wait_for_completion();
285 }
286}

288template <typename I, typename O, typename Of, typename Ip, typename Op>
289void
290WorkDispatcher<I, O, Of, Ip, Op>::thread_pool_main(const Device & device)
291{
292 while (true)
293 {
294 std::function<void()> task;
295 {
296 std::unique_lock<std::mutex> lock(_qmutex);
297 _thread_condition.wait(lock, [this, &device] { return _stop || !_tasks.at(device).empty(); });
298 if (_stop && _tasks.at(device).empty())
299 break;
300 task = std::move(_tasks.at(device).front());
301 _tasks.at(device).pop();
302 }
303 task();
304 }
305}

307template <typename I, typename O, typename Of, typename Ip, typename Op>
308void
309WorkDispatcher<I, O, Of, Ip, Op>::stop_thread_pool()
310{
311 if (!_async)
312 return;
313 {
314 std::unique_lock<std::mutex> lock(_qmutex);
315 _stop = true;
316 }
317 _thread_condition.notify_all();
318 for (auto & thread : _thread_pool)
319 thread.join();
320}

351template <typename I, typename O, typename Of, typename Ip, typename Op>
352Of
353WorkDispatcher<I, O, Of, Ip, Op>::run_sync(WorkGenerator<Ip> & generator)
354{
355 validate();
356
357 Device device = kCPU;
358 std::size_t n = 0;
359 _results.clear();
360 while (generator.has_more())
361 {
362 _scheduler.schedule_work(device, n);
363 if (n <= 0)
364 throw NEMLException("Scheduler returned a batch size of " + std::to_string(n));
365 // Generate work
366 auto && [m, work] = generator.next(n);
367 // Preprocess
368 if (_preprocess)
369 work = _preprocess(std::move(work), device);
370 // Do work. Since there is no asynchronous execution, we do not notify the scheduler (this also
371 // avoids potential parallel communication inccured by the scheduler)
372 auto result = _do_work(std::move(work), device);
373 // Postprocess
374 if (_postprocess)
375 result = _postprocess(std::move(result));
376 _results.push_back(result);
377 }
378
379 if (_reduce)
380 return _reduce(std::move(_results));
381
382 if constexpr (std::is_same<Of, std::vector<Op>>::value)
383 return _results;
384
385 throw NEMLException("Internal error: unreachable code");
386}

388template <typename I, typename O, typename Of, typename Ip, typename Op>
389Of
390WorkDispatcher<I, O, Of, Ip, Op>::run_async(WorkGenerator<Ip> & generator)
391{
392 validate();
393
394 Device device = kCPU;
395 std::size_t n = 0;
396 _results.clear();
397
398 // Keep asking the scheduler for an available device
399 // - If the generator has no more work, we break out of the loop
400 // - If the scheduler schedules work, we dispatch the work and continue with the dispatching loop
401 while (generator.has_more())
402 {
403 _scheduler.schedule_work(device, n);
404 if (n <= 0)
405 throw NEMLException("Scheduler returned a batch size of " + std::to_string(n));
406 // Generate work
407 auto && [m, work] = generator.next(n);
408 // Reserve space for the result
409 _results.resize(_results.size() + 1);
410 auto i = _results.size() - 1;
411 // Create the task
412 auto task = [this, work = std::move(work), device = device, m = m, i = i]() mutable
413 {
414 // Preprocess
415 if (_preprocess)
416 work = _preprocess(std::move(work), device);
417 // Do work
418 auto result = _do_work(std::move(work), device);
419 // Postprocess
420 if (_postprocess)
421 result = _postprocess(std::move(result));
422 // Collect result
423 _results[i] = std::move(result);
424 // Tell the scheduler that we have completed m batches
425 _scheduler.completed_work(device, m);
426 };
427 // Tell the scheduler that we have dispatched m batches
428 _scheduler.dispatched_work(device, m);
429 // Enqueue the task
430 {
431 std::lock_guard<std::mutex> lock(_qmutex);
432 _tasks.at(device).push(task);
433 }
434 // Notify the thread pool
435 // Note: We notify_all instead of notify_one because we want the thread that's bind to the
436 // target device to pick up the task
437 _thread_condition.notify_all();
438 }
439
440 // Wait for all tasks to complete
441 _scheduler.wait_for_completion();
442
443 if (_reduce)
444 return _reduce(std::move(_results));
445
446 if constexpr (std::is_same<Of, std::vector<Op>>::value)
447 return _results;
448
449 throw NEMLException("Internal error: unreachable code");
450}
