259 template <typename T = Model, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
260 T & register_model(const std::string & name, bool merge_input = true)
261 {
262 auto model_name =
263 input_options().contains(name) ? input_options().get<std::string>(name) : name;
264 if (model_name == this->name())
265 throw SetupException("Model named '" + this->name() +
266 "' is trying to register itself as a sub-model. This is not allowed.");
267
268 OptionSet extra_opts;
269 extra_opts.set<NEML2Object *>("_host") = host();
270
271 if (!host()->factory())
272 throw SetupException("Internal error: Host object '" + host()->name() +
273 "' does not have a factory set.");
274 auto model = host()->factory()->get_object<T>("Models", model_name, extra_opts);
275
276 register_model(model, merge_input);
277 return *model;
278 }

291 template <typename T = Model, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
292 void register_model(const std::shared_ptr<T> & model, bool merge_input = true)
293 {
294 if (std::find(_registered_models.begin(), _registered_models.end(), model) !=
295 _registered_models.end())
296 throw SetupException("Model named '" + model->name() + "' has already been registered.");
297
298 if (merge_input)
299 for (auto && [name, var] : model->input_variables())
300 clone_input_variable(*var);
301
302 _registered_models.push_back(model);
303 }
