159template <class Derived, typename>
160Derived
161CrystalGeometry::slip_slice(const Derived & tensor, Size grp) const
162{
163 if (grp >= nslip_groups())
164 throw NEMLException("Invalid slip group index");
165 return tensor.batch_index(
166 {indexing::Ellipsis, indexing::Slice(_slip_offsets[grp], _slip_offsets[grp + 1])});
167}
