71template <typename T>
72void
73parse_(T & val, const std::string & raw_str)
74{
75 std::stringstream ss(trim(raw_str));
76 ss >> val;
77 if (ss.fail())
78 throw ParserException("Failed to parse '" + raw_str + "' as a " +
79 utils::demangle(typeid(T).name()));
80}

82template <typename T>
83T
84parse(const std::string & raw_str)
85{
86 T val;
87 parse_(val, raw_str);
88 return val;
89}

91template <typename T>
92void
93parse_vector_(std::vector<T> & vals, const std::string & raw_str)
94{
95 auto tokens = split(raw_str, " \t\n\v\f\r");
96 vals.resize(tokens.size());
97 for (size_t i = 0; i < tokens.size(); i++)
98 parse_<T>(vals[i], tokens[i]);
99}
