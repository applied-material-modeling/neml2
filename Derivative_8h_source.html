37template <std::size_t N>
38std::string
39derivative_name(const std::string & var_name, const std::array<std::string, N> & arg_names)
40{
41 std::string name = "d";
42 if constexpr (N > 1)
43 name += std::to_string(N);
44 name += "(" + var_name + ")/";
45 for (const auto & arg_name : arg_names)
46 name += "d(" + arg_name + ")";
47 return name;
48}
